-- se eliminan archivos en hdfs

fs -rm -r -f output
fs -rm -r -f *.csv

-- copia de archivos del sistema local al HDFS

fs -put data.csv

-- carga de archivos


u = LOAD 'data.csv' USING PigStorage(',') 
    AS (id:int, 
        firstname:CHARARRAY, 
        surname:CHARARRAY, 
        birthday:CHARARRAY, 
        color:CHARARRAY, 
        quantity:INT);


-- columna de interés

uu = FOREACH u GENERATE firstname, color;

-- filtro - resultado

z = FILTER uu BY ((ENDSWITH(color, 'a') == true) 
              OR (ENDSWITH(color, 'e') == true) 
              OR (ENDSWITH(color, 'i') == true)
              OR (ENDSWITH(color, 'o') == true)
              OR (ENDSWITH(color, 'u') == true));

-- almacenamiento

STORE z INTO 'output' USING PigStorage(',');


-- copia de archivos del HDFS al sistema local

fs -get output/ ;
